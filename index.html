🔗 Synchronisation Cloud

Partagez les données en temps réel avec toute l'équipe via Google Sheets

⚪ Aucune URL configurée. Entrez l'URL de votre script Google Apps Script.
💡
Comment ça marche ?
1. Créez un Google Sheet et collez-y le script Apps Script fourni.
2. Déployez le script en tant qu'application web (accès : Tout le monde).
3. Copiez l'URL de déploiement et collez-la ici, puis cliquez sur "Enregistrer & Tester".
4. Toutes les saisies seront automatiquement partagées avec l'équipe !

📋 Script Google Apps Script

Copiez ce code dans votre Google Sheet (Extensions → Apps Script)

function doGet(e){
  var s=SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Transactions');
  if(!s)return resp([]);
  var d=s.getDataRange().getValues();
  if(d.length<=1)return resp([]);
  var h=d[0],r=[];
  for(var i=1;i<d.length;i++){
    var o={};for(var j=0;j<h.length;j++)o[h[j]]=d[i][j];
    o.id=Number(o.id);o.montant=Number(o.montant);o.ts=Number(o.ts);r.push(o);
  }
  return resp(r);
}
function doPost(e){
  var ss=SpreadsheetApp.getActiveSpreadsheet();
  var s=ss.getSheetByName('Transactions');
  if(!s){s=ss.insertSheet('Transactions');
    s.appendRow(['id','date','type','compte','payer','evt','cat','montant','desc','remb','ts']);}
  var p=JSON.parse(e.postData.contents);
  if(p.action==='add'){
    var x=p.entry;
    s.appendRow([x.id,x.date,x.type,x.compte,x.payer,x.evt,x.cat,x.montant,x.desc,x.remb,x.ts]);
  }else if(p.action==='delete'){
    var d=s.getDataRange().getValues();
    for(var i=1;i<d.length;i++){if(Number(d[i][0])===Number(p.id)){s.deleteRow(i+1);break;}}
  }else if(p.action==='sync_all'){
    s.clearContents();
    s.appendRow(['id','date','type','compte','payer','evt','cat','montant','desc','remb','ts']);
    var ex=[...p.entries].reverse();
    ex.forEach(function(x){s.appendRow([x.id,x.date,x.type,x.compte,x.payer,x.evt,x.cat,x.montant,x.desc,x.remb,x.ts]);});
  }
  return resp({status:'success'});
}
function resp(d){
  return ContentService.createTextOutput(JSON.stringify(d))
    .setMimeType(ContentService.MimeType.JSON);
}
